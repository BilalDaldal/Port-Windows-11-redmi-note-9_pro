#####push parted script to sdcard######
adb push parted /sdcard/

####start adb shell#####
adb shell

#####move parted to sbin directory###
cp /sdcard/parted /sbin/ && chmod 755 /sbin/parted

#unmount data and sdcard
umount /data && umount /sdcard

#start parted
parted /dev/block/sda

#Remove partitions rm {partition number} #
rm30

room 29

rm 28

#####Restore partitions#########

#128Gb models
mkpart userdata ext4 11.8GB 127GB

#256Gb models
mkpart userdata ext4 11.8GB 255GB

#Quit
quit
