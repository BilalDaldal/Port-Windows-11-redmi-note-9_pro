Este proceso hay que realizarlo después de haber hecho los comandos en termux

Iniciar TWRP recovery a través el PC con el comando 
fastboot boot <twrp.img>

No copiar las líneas con #

#####push parted script to sdcard######
adb push parted /sdcard/

####iniciar adb shell#####
adb shell

#####mover parted a directorio sbin###
cp /sdcard/parted /sbin/ && chmod 755 /sbin/parted

#desmontar data y sdcard
umount /data && umount /sdcard

#Iniciar parted
parted /dev/block/sda

#Eliminar particiones rm {número partición} #
rm 32

#####Crear particiones#########

#Crear particiones (modelos de 256Gb, Aún no está disponible para el de 128Gb)
mkpart esp fat32 11.8GB 12.2GB
mkpart win ntfs 12.2GB 92.2GB
mkpart userdata ext4 92.2GB 255GB


#set esp on
set 28 esp on

#Quit
quit

#reiniciar a recovery twrp

#formatea las nuevas particiones

mkfs.fat -F32 -s1 /dev/block/by-name/esp

mkfs.ntfs -f /dev/block/by-name/win

#Formatear desde twrp 

#Comprueba si inicia Android

##con estos comandos tu pc podra detectar el movil como un disco para llevar a cabo la instalacion (Tener desactivado el MTP en Mount)

#Pasar msc.sh a la memoria interna
adb push msc.sh /sdcard

#iniciar shell
adb shell

#indicar que queremos usar la raiz
cd /sdcard

#Estos dos comandos arrancan el programa 
chmod +x msc.sh

./msc.sh

###Una vez se detecte al X3 pro como disco###

#abrimos un cmd 

#Esto arrancará la herramienta de administración de discos de Windows 
diskpart

#Selecciona el disco que sea tu x3 pro
sel disk 1 (o el que sea tu poco x3 pro)

#Seleccionamos el volumen donde vamos a instalar Windows 
sel vol 3 

#Le asignamos la letra x al volumen donde vamos a instalar Windows 
assign letter=x

#Seleccionamos el volumen donde instalaremos los drivers
sel vol 4 

#Le asignamos la letra y al volumen donde instalaremos los drivers
assign letter=y

#Instalar Windows 11 en el X3 Pro (sustituir "Usuario" por tu usuario y "Carpetadewindowsarm"por tu carpeta que contiene windows arm,importante que sea sin espacios)
dism /apply-image /ImageFile:"C:\Users\Usuario\Downloads\Carpetadewindowsarm\sources\install.wim" /index:1 /ApplyDir:X:\ 

#Instalar drivers (sustituir mipad5folder por la ubicacion de la carpeta de los drivers)(abrir en powershell como administrador)
.\driverupdater.exe -d (mipad5folder)\definitions\Desktop\ARM64\Internal\vayu.txt -r (mipad5folder) -p X:

#Ahora se tienen que permitir los drivers no firmados
bcdboot X:\Windows /s Y: /f UEFI

bcdedit /store Y:\EFI\Microsoft\BOOT\BCD /set {default} testsigning on

#Este comando es opcional,es para que salga un menú de inicio
bcdedit /store Y:\EFI\Microsoft\BOOT\BCD /set {bootmgr} displaybootmenu yes


###############Inicia con la imagen UEFI de arranque de Windows##############

fastboot flash boot (Uefi del dispositivo.img)
