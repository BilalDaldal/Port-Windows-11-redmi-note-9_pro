#desmontar data y sdcard
umount /data && umount /sdcard

#Iniciar parted

#Crear particiones (256GB model,si tienes el de 128gb usa el comando de iniciar parted y luego pones P para ver las  particiones,pásame captura de eso y haré los comandos)
mkpart esp fat32 11.7GB 12.1GB
mkpart win ntfs 12.1GB 92.1GB
mkpart pe fat32 92.1GB 95.1GB

set 32 esp on
